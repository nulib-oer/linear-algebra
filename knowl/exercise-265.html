\begin{equation*} A=cI_n=\begin{amatrix}[cccc] c \amp 0\amp \dots\amp 0\\ 0\amp c\amp \dots \amp 0 \\ \vdots \amp \amp \amp \vdots\\ 0\amp 0\amp \dots \amp c \end{amatrix}\text{.} \end{equation*}
