\begin{equation*} W=\text{span} \lbrace \left[\begin{array}{cc} -2 \amp 4\cr 4 \amp 3 \end{array}\right],\left[\begin{array}{cc} 4 \amp 3\cr 3 \amp 2 \end{array}\right] \rbrace . \end{equation*}
